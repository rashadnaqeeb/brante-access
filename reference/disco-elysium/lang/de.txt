# Gedankenkabinett, thought = Gedanke, thought slot = Platz / Gedankenplatz, skill =

# Auftrag, signature skill = Spezialfähigkeit, locked = gesperrt, currency = Reál,
# experience = Erfahrung, research time = Forschungszeit, remaining = verbleibend,
# bullets = Kugeln, clothing = Kleidung.
# Place names taken from the game's own German text: cargo container = Frachtcontainer,
# secretary's office = Sekretariat, union = Gewerkschaft, buoy = Boje, dumpster =
# Müllcontainer, trash can = Mülltonne, metal box = Metallkiste, canister = Kanister.
# The mod's "orb" is Orb, not Kugel: the game already calls the inventory's bullets Kugeln.
# A lootable container is a Behälter; Container alone is a freight/refuse container.
# Words spoken after a noun the game picks (an item, a category, a health bar) stay
# predicative, which German does not inflect: leer, neu, kritisch, offen, nichts.
